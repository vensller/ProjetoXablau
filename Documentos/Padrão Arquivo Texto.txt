//Tudo que tiver // será desconsiderado
//Deve ser separado por ; e pulado para a próxima linha
//Tudo o que estiver depois do ';' e na mesma linha será desconsiderado
//Podem estar embaralhados os vértices, arestas e indivíduos
//O padrão para vértice, aresta e indivíduo deve ser o seguinte:

//Vértices (1º: Rótulo, 2º X, 3º Y) 
v:A,1,1;
v:B,2,3;
v:C,3,4;
v:D,5,7;

//Arestas (1º: Vértice origem, 2º Vértice Destino, 3º Comprimento, 4º nome)
a:A,B,5, nomeA;
a:B,C,4, nomeB;
a:D,A,7, nomeC;
a:A,C,8, nomeD;
a:A,D,7, nomeE;

//Indivíduos (1º Vértice de localização, 2º nome)
i:A, nomeA;
i:B, nomeB;
i:D, nomeC;
